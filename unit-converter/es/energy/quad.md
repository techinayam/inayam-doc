\[ \text{Energy in quads} = \frac{\text{Energy in BTUs}}{1,000,000,000,000,000} \]

\[ \text{Energy in quads} = \frac{2,000,000,000,000}{1,000,000,000,000,000} = 0.002 \text{ quads} \]
