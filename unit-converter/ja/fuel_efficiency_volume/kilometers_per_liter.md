\[ \text{Fuel Efficiency (km/L)} = \frac{\text{Distance Traveled (km)}}{\text{Fuel Used (L)}} \]

\[ \text{Fuel Efficiency (km/L)} = \frac{300 \text{ km}}{20 \text{ L}} = 15 \text{ km/L} \]
