\[ \text{Angular Acceleration} (\alpha) = \frac{\Delta \omega}{\Delta t} \]

\[ \alpha = \frac{6 \, \text{rad/s} - 2 \, \text{rad/s}}{2 \, \text{s}} = 2 \, \text{rad/s}^3 \]
